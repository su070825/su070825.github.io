<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>温馨提醒！！</title>
    <style>
        /* 弹窗样式：固定定位、悬浮在页面上方 */
        .tip-box {
            position: fixed;
            width: 280px;
            height: 60px;
            padding: 10px;
            background: lightblue;
            border-radius: 8px;
            box-shadow: 0 2px 8px rgba(0,0,0,0.2);
            display: flex;
            align-items: center;
            justify-content: center;
            font-family: "微软雅黑", sans-serif;
            font-size: 14px;
            z-index: 9999; /* 确保弹窗在最上层 */
        }
    </style>
</head>
<body>
<script>
    const tips = ["要天天开心！！", "不可以不吃饭！", "我很想很想你", "我想见面", "要放假啦", "要早点睡觉！！", "一切都会变好哒", "明年一定见", "我比你想象的还要在乎你", "要好好照顾自己呢！", "平安",
        "要相信自己嗷", "我在天津很想你", "想我了嘛", "顺顺利利！！", "烦恼消失！！"];
    const bgColors = ["skyblue", "lightgreen", "lightyellow", "pink"];
    const tipElements = [];
    function createTip() {
        const tipBox = document.createElement("div");
        tipBox.className = "tip-box";
        const tip = tips[Math.floor(Math.random() * tips.length)];
        const bgColor = bgColors[Math.floor(Math.random() * bgColors.length)];
        tipBox.textContent = tip;
        tipBox.style.background = bgColor;
        const maxX = window.innerWidth - tipBox.offsetWidth;
        const maxY = window.innerHeight - tipBox.offsetHeight;
        const x = Math.floor(Math.random() * maxX);
        const y = Math.floor(Math.random() * maxY);
        tipBox.style.left = `${x}px`;
        tipBox.style.top = `${y}px`;
        document.body.appendChild(tipBox);
        tipElements.push(tipBox);
        }
    function closeAllTips() {
            tipElements.forEach(el => el.remove());
        }
    document.addEventListener("keydown", (e) => {
        if (e.code === "Space") {
            closeAllTips();
        }
    });
    for (let i = 0; i < 400; i++) {
        setTimeout(createTip, i * 50);
    }
    </script>
</body>
</html>
